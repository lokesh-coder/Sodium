105:      *

108:      * @param int $itemsPerPage The maximum number of items to display on a page, in the range 1..100. Default is 20.

231:     /**
232:      * Retrieves data from the Kuler API
233:      *
234:      * @param string $endpoint The API endpoint to query against.
235:      * @param Array $params Parameters that will be parsed by http_build_query and appended to the URL.
236:      * @uses http_build_query()
237:      * @uses SimpleXmlElement
238:      * @see self::URL_* constants
239:      * @return SimpleXmlElement The XML response parsed into a SimpleXmlElement.
240:      */
241:     protected function request($endpoint, $params)
242:     {
243:         if (isset($params['itemsPerPage']) && ($params['itemsPerPage'] > 100 || $params['itemsPerPage'] < 1))
244:             throw new Kuler_Exception('The number of items per page must be between 1 and 100.');
245: 
246:         $params = array_merge(array(
247:             'key' => $this->key,
248:         ), $params);
249: 
250:         try {
251:             $this->response = new SimpleXmlElement(self::buildUrl($endpoint, $params), NULL, true);
252:         } catch (Exception $e) {
253:             throw new Kuler_Exception('Error retrieving the feed: ' . $e->getMessage());
254:         }
255:         return $this->response;
256:     }
